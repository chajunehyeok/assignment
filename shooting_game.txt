typedef struct {
    int x, y;
    int alive;
    int direction;  // 1: 오른쪽으로 이동, -1: 왼쪽으로 이동
    Bullet bullets[MAX_ENEMY_BULLETS];  // 적의 총알
} Enemy;

void initializeGame() {
for (int i = 0; i < MAX_ENEMIES; i++) {
        enemies[i].x = rand() % (SCREEN_WIDTH - 2);
        enemies[i].y = rand() % (SCREEN_HEIGHT / 2);
        enemies[i].alive = 1;
        enemies[i].direction = (rand() % 2) == 0 ? 1 : -1;
}
